1. Sum of N Numbers

Write a C program that takes an integer `N` from the user and calculates the sum of all numbers from 1 to `N`.

Expected Output (for N=5):

Enter a positive integer N: 5
Sum of numbers from 1 to 5 is: 15

Sample Solution:

#include <stdio.h>

int main() {
    int N, sum = 0;
    printf("Enter a positive integer N: ");
    scanf("%d", &N);

    if (N <= 0) {
        printf("Please enter a positive integer.\n");
        return 1;
    }

    for (int i = 1; i <= N; i++) {
        sum += i;
    }

    printf("Sum of numbers from 1 to %d is: %d\n", N, sum);
    return 0;
}

2. Reverse a String

Write a C program to reverse a given string without using any built-in string reversal functions.

Expected Output (for input "hello"):

Enter a string: hello
Reversed string: olleh

Sample Solution:

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    char reversed_str[100];
    int length, i, j;

    printf("Enter a string: ");
    scanf("%s", str);

    length = strlen(str);
    j = 0;
    for (i = length - 1; i >= 0; i--) {
        reversed_str[j] = str[i];
        j++;
    }
    reversed_str[j] = '\0'; // Null-terminate the reversed string

    printf("Reversed string: %s\n", reversed_str);
    return 0;
}

3. Find Largest Element in Array

Write a C program to find the largest element in an array of integers.

Expected Output (for array {12, 45, 9, 78, 34}):

The largest element is: 78

Sample Solution:

#include <stdio.h>

int main() {
    int arr[] = {12, 45, 9, 78, 34};
    int n = sizeof(arr) / sizeof(arr[0]);
    int largest = arr[0];

    for (int i = 1; i < n; i++) {
        if (arr[i] > largest) {
            largest = arr[i];
        }
    }

    printf("The largest element is: %d\n", largest);
    return 0;
}

4. Check Palindrome

Write a C program to check if a given string is a palindrome (reads the same forwards and backwards).

Expected Output (for input "madam"):

Enter a string: madam
madam is a palindrome.

Sample Solution:

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int length, i, j, is_palindrome = 1;

    printf("Enter a string: ");
    scanf("%s", str);

    length = strlen(str);
    i = 0;
    j = length - 1;

    while (i < j) {
        if (str[i] != str[j]) {
            is_palindrome = 0;
            break;
        }
        i++;
        j--;
    }

    if (is_palindrome) {
        printf("%s is a palindrome.\n", str);
    } else {
        printf("%s is not a palindrome.\n", str);
    }
    return 0;
}

5. Fibonacci Series

Write a C program to generate the first `N` terms of the Fibonacci series.

Expected Output (for N=7):

Enter the number of terms: 7
Fibonacci Series: 0 1 1 2 3 5 8

Sample Solution:

#include <stdio.h>

int main() {
    int n, t1 = 0, t2 = 1, nextTerm;

    printf("Enter the number of terms: ");
    scanf("%d", &n);

    printf("Fibonacci Series: ");

    for (int i = 1; i <= n; ++i) {
        printf("%d ", t1);
        nextTerm = t1 + t2;
        t1 = t2;
        t2 = nextTerm;
    }
    printf("\n");
    return 0;
}

6. Swap Two Numbers

Write a C program to swap two numbers using pointers.

Expected Output (for inputs 5 and 10):

Enter value for num1: 5
Enter value for num2: 10
Before swap: num1 = 5, num2 = 10
After swap: num1 = 10, num2 = 5

Sample Solution:

#include <stdio.h>

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int num1, num2;
    printf("Enter value for num1: ");
    scanf("%d", &num1);
    printf("Enter value for num2: ");
    scanf("%d", &num2);

    printf("Before swap: num1 = %d, num2 = %d\n", num1, num2);
    swap(&num1, &num2); // Pass addresses
    printf("After swap: num1 = %d, num2 = %d\n", num1, num2);
    return 0;
}

7. Count Vowels and Consonants

Write a C program to count the number of vowels and consonants in a given string.

Expected Output (for input "Programming"):

Enter a string: Programming
Vowels: 3
Consonants: 8

Sample Solution:

#include <stdio.h>
#include <string.h>
#include <ctype.h> // For tolower()

int main() {
    char str[100];
    int vowels = 0, consonants = 0;

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin); // Read string including spaces
    str[strcspn(str, "\n")] = 0; // Remove trailing newline

    for (int i = 0; str[i] != '\0'; i++) {
        char ch = tolower(str[i]);
        if (ch >= 'a' && ch <= 'z') { // Check if it's an alphabet
            if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
                vowels++;
            } else {
                consonants++;
            }
        }
    }

    printf("Vowels: %d\n", vowels);
    printf("Consonants: %d\n", consonants);
    return 0;
}

8. Simple File Read/Write

Write a C program to create a text file, write some content to it, and then read and display the content.

Expected Output (console output, also creates `sample.txt`):

File 'sample.txt' written successfully.
Content of 'sample.txt':
Hello, BOZON ACADEMY!
This is a test line.

Sample Solution:

#include <stdio.h>
#include <stdlib.h> // For exit()

int main() {
    FILE *fptr;
    char data[100];

    // Writing to a file
    fptr = fopen("sample.txt", "w"); // Open in write mode
    if (fptr == NULL) {
        printf("Error opening file for writing!\n");
        exit(1);
    }
    fprintf(fptr, "Hello, BOZON ACADEMY!\n");
    fprintf(fptr, "This is a test line.\n");
    fclose(fptr);
    printf("File 'sample.txt' written successfully.\n");

    // Reading from a file
    fptr = fopen("sample.txt", "r"); // Open in read mode
    if (fptr == NULL) {
        printf("Error opening file for reading!\n");
        exit(1);
    }
    printf("Content of 'sample.txt':\n");
    while (fgets(data, 100, fptr) != NULL) {
        printf("%s", data);
    }
    fclose(fptr);
    return 0;
}

9. Implement `strlen()`

Write your own version of the `strlen()` function (string length) without using `string.h`.

Expected Output (for input "Bozon"):

Enter a string: Bozon
Length of "Bozon" is: 5

Sample Solution:

#include <stdio.h>

int my_strlen(const char *str) {
    int length = 0;
    while (str[length] != '\0') {
        length++;
    }
    return length;
}

int main() {
    char text[100];
    printf("Enter a string: ");
    scanf("%s", text);
    printf("Length of \"%s\" is: %d\n", text, my_strlen(text));
    return 0;
}

10. Bubble Sort

Implement the Bubble Sort algorithm to sort an array of integers.

Expected Output (for array {64, 34, 25, 12, 22, 11, 90}):

Original array: 64 34 25 12 22 11 90 
Sorted array: 11 12 22 25 34 64 90

Sample Solution:

#include <stdio.h>

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j+1]) {
                // Swap arr[j] and arr[j+1]
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}

void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr) / sizeof(arr[0]);

    printf("Original array: ");
    printArray(arr, n);

    bubbleSort(arr, n);

    printf("Sorted array: ");
    printArray(arr, n);
    return 0;
}
